创建数据库：

用户表：

```sql
CREATE table User(
	id BIGINT(30) NOT NULL AUTO_INCREMENT,
	uid BIGINT(50)	NOT NULL COMMENT = '账号id',
	sex tinyint(4) DEFAULT '0' COMMENT '性别：0男   1女',
	vipId BIGINT(50) DEFAULt null	COMMENT '会员id',
	userName VARCHAR(30) not null COMMENT = '用户名',
	password VARCHAR(30) not NULL COMMENT = '密码',
	mail VARCHAR(30) DEFAULT NULL COMMENT = '邮箱',
	phone VARCHAR(30) DEFAULT NULL COMMENT = '手机号',
	wxId VARCHAR(30) DEFAULT NULL COMMENT = '微信标识',
	extra text COMMENT = '额外字段',
	create_person varchar(60) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci DEFAULT NULL COMMENT '创建人',
  create_time datetime DEFAULT NULL COMMENT '创建日期',
  update_person varchar(60) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci DEFAULT NULL COMMENT '更新人',
  update_time datetime DEFAULT NULL COMMENT '更新日期',
  dr tinyint(4) DEFAULT '0' COMMENT '删除标识：0有效   1删除',
	PRIMARY KEY (id) USING BTREE,
	index phone (phone),
  index uid (uid),
	index userName (userName),
	index mail (mail),
	index wxId (wxId)
)ENGINE = INNODB CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci COMMENT '账号表' ROW_FORMAT = Dynamic;
```
